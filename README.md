## Install

```bash
npm install --save brandswap-ds
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'brandswap-ds'
import 'brandswap-ds/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
